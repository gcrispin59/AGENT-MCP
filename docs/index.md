## Quick Start

```bash
# Install AGENT MCP
npm install agent-mcp

# Create a basic agent
const { BaseAgent } = require('agent-mcp');

const agent = new BaseAgent({
    type: 'worker',
    capabilities: ['processing']
});

await agent.initialize();
```
